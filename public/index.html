How to Use

Web3.js

const Web3 = require('web3');
const web3 = new Web3('https://rpc.celo-community.org');

ethers.js

const { ethers } = require('ethers');
const provider = new ethers.providers.JsonRpcProvider('https://rpc.celo-community.org');

celo-cli

celocli node:synced --node https://rpc.celo-community.org
